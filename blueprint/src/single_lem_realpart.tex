\chapter{lem:realpart}\label{chap:lem_realpart}
\begin{lemma}[Positivity of $\mathrm{Re}\,\psi$]
\label{lem:realpart}
\lean{LeaHadamard.Hadamard.re_psi_taylor4_decomposition}
With $\psi$ as in the cosine-product bound and
$s(\lambda) := \sum_{e}\lambda_e^2$, if $s(\lambda) \le 1/2$, then
$\mathrm{Re}\,\psi(\lambda) \in [3/4, 1]$.
\end{lemma}
