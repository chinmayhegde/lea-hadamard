\chapter{Lambda facts}
\label{chap:lambda-facts}

\begin{lemma}[Multiplicativity of $\psi$ under $\Lambda$-translation]
\label{lem:psi-translate}
\lean{LeaHadamard.Hadamard.psi_translate_lambdaReal}
For every $\lambda \in \Lambda$ and every $\gamma \in \Bbox_{\pi/4}
:= (-\pi/4,\,\pi/4)^d$,
\[
  \psi(\lambda + \gamma) \;=\; \psi(\lambda)\,\psi(\gamma).
\]
\end{lemma}

\begin{lemma}[Disjointness of $\Lambda$-translated boxes]
\label{lem:disjoint-boxes}
\lean{LeaHadamard.Hadamard.translated_edgeBoxes_disjoint}
Let $0 < \delta < \pi/4$. For each $\lambda \in \Lambda$, let
$\Bbox_\delta(\lambda) := \lambda + (-\delta,\,\delta)^d \subset \mathbb{T}^d$.
Then the family $\{\Bbox_\delta(\lambda)\}_{\lambda \in \Lambda}$ is
pairwise disjoint: for any two distinct $\lambda, \lambda' \in \Lambda$,
\[
  \Bbox_\delta(\lambda) \cap \Bbox_\delta(\lambda') \;=\; \emptyset.
\]
\end{lemma}

\begin{lemma}[Cardinality of $\Lambda$ and unit-modulus phase]
\label{lem:lambda-card}
\lean{LeaHadamard.Hadamard.card_lambdaShifts_eq_pow_of_two_le}
Let $n \ge 2$ and $d = \binom{n}{2}$. The lattice $\Lambda$ has cardinality
\[
  |\Lambda| \;=\; 2^{\,2d - n + 1},
\]
and for every integer $t \ge 1$ and every $\lambda \in \Lambda$,
\[
  \psi(\lambda)^{4t} \;=\; 1.
\]
\end{lemma}

\begin{lemma}[Even-degree characterization of $\Lambda$]
\label{lem:even-degree}
\lean{LeaHadamard.Hadamard.parityEvenBool_iff_incidence_zero}
Let $\lambda \in \Lambda_0$ and let $\lambda^{(2)} \in \{0,1\}^d$ be its
mod-2 reduction, viewed as the edge-indicator of a subgraph
$G_{\lambda^{(2)}}$ of the complete graph $K_n$. Then
\[
  \lambda \in \Lambda
  \;\Longleftrightarrow\;
  \text{every vertex } k \in \{1,\dots,n\} \text{ has even degree in }
  G_{\lambda^{(2)}}.
\]
\end{lemma}
